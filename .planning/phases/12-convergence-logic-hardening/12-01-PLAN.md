```javascript
function computeEMA(totals, alpha) {
  if (totals.length === 0) return [];
  var ema = [totals[0]];
  for (var i = 1; i < totals.length; i++) {
    ema.push(alpha * totals[i] + (1 - alpha) * ema[i - 1]);
  }
  return ema;
}
```
